00042     bool is_hermitian(const matrix<T,A>& a)
00043     {
00044         if(num_rows(a) != num_cols(a))
00045             return false;
00046         for (size_t i=0; i < num_rows(a); ++i)
00047             for(size_t j=0; j < num_cols(a); ++j){
00048                if( a(i,j) != ambient::numeric::kernels::helper_complex<T>::conj(a(j,i)) )
00049                    return false;
00050             }
00051         return true;
