00058     template<typename T>
00059     class binary_tree {
00060     public:
00061         binary_tree(size_t N) : tree(N), length(N) {
00062             entry_point = N/2;
00063             generate(tree, N, N);
00064             normalize();
00065             check();
00066         }
00067         T generate(std::vector<std::pair<T,T> >& tree, int N, int L, int start = 0){
00068             if(N <= 0 || start >= L) return BOUNDARY_OVERFLOW;
00069             tree[start+N/2] = std::make_pair(generate(tree, N/2, L, start), 
00070                                              generate(tree, N-N/2-1, L, start+N/2+1));
00071             return start+N/2;
00072         }
00073         void normalize(){
00074             std::vector<std::pair<T,T> > normalized(length);
00075             for(int i = 0; i < length; i++){
00076                 normalized[i] = tree[(i+entry_point)%length];
00077                 if(normalized[i].first  != BOUNDARY_OVERFLOW) normalized[i].first  = (normalized[i].first  - entry_point + length) % length;
00078                 if(normalized[i].second != BOUNDARY_OVERFLOW) normalized[i].second = (normalized[i].second - entry_point + length) % length;
00079             }
00080             std::swap(tree, normalized);
00081             entry_point = 0;
00082         }
00083         void check(){
00084             std::vector<bool> states(length);
00085             for(int i = 0; i < length; i++){
00086                 if(tree[i].first  != BOUNDARY_OVERFLOW) states[tree[i].first]  = true;
00087                 if(tree[i].second != BOUNDARY_OVERFLOW) states[tree[i].second] = true;
00088             }
00089             for(int i = 0; i < length; i++){
00090                 if(!states[i] && i != entry_point) throw std::runtime_error("Error: no route to host");
00091             }
00092         }
00093         std::pair<T,T> operator[](int i) const {
00094             return tree[i];
00095         }
00096     private:
00097         std::vector<std::pair<T,T> > tree;
00098         int entry_point;
00099         size_t length;
00100     };
