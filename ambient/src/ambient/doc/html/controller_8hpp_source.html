00048     inline void controller::flush(){
00049         AMBIENT_SMP_ENABLE
00050         while(!chains->empty()){
00051             for(auto task : *chains){
00052                 if(task->ready()){
00053                     AMBIENT_THREAD task->invoke();
00054                     for(auto d : task->deps) d->ready();
00055                     mirror->insert(mirror->end(), task->deps.begin(), task->deps.end());
00056                 }else mirror->push_back(task);
00057             }
00058             chains->clear();
00059             std::swap(chains,mirror);
00060         }
00061         AMBIENT_SMP_DISABLE
00062         model.clock++;
00063         fence();
00064     }
